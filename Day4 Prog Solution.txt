def max_num(n1,n2,n3):
    if n1>=n2 and n1>=n3:
        print("Max number is:",n1)
    elif n2>=n1 and n2>=n3:
        print("Max number is:",n2)
    else:
        print("Max number is:",n3)
max_num(3,1,9)

3.Write a recursive function to calculate the sum of numbers from 0 to 10
Expected output: 55
Ans:
def recursiveSum(n): 
    if n <= 1: 
        return n 
    return n + recursiveSum(n-1) 
print(recursiveSum(10))
