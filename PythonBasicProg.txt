#Q:Program to find number is even or odd.
num=int(input("enter the number"))
if num%2==0:
    print(num,"is even")
else:
    print(num,"is odd")


#Q:Program to find number is even or odd without using modulus operator.
num=int(input("enter the number"))
def iseven(num):
    if (int(num/2)*2==num):
        print(num,"is even")
    else:
        print(num,"is odd")
iseven(num)


#Q:Program to find factorial of number.
num=int(input("enter the number"))
f=1
for i in range(1,num+1):
    f=f*i
print("factorial of",num,"is",f)


#Q:Program to find factorial of number using recursion.
num=int(input("enter the number"))
def factorial(num):
    f=1
    for i in range(1,num+1):
        f=f*i
    print("factorial of",num,"is",f)
factorial(num)


#Q:Program to find number is prime or not.
num=int(input("enter the number"))
if num>1:
    for i in range(2,num):
        if num%i==0:
            print(num,"is not prime")
            break
    else:
        print(num,"is prime")
else:
    print(num,"is not prime")
