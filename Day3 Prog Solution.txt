2.From a list containing ints, strings and floats, make three lists to store them separately. 
list1=[11,22,"Java",33,44.7,"Python",20.3]
int_item=[]
float_item=[]
str_item=[]
for i in list1:
    if type(i)==int:
        int_item.append(i)
    elif type(i)==str:
        str_item.append(i)
    else:
        float_item.append(i)
print("Integer item list:",int_item)
print("String item list:",str_item)    
print("Float item list:",float_item)

4. Accept data in two 3*3  matrix and calculate the sum of the matrices
Ans:
import numpy as np
r=int(input("Enter the row size"))
c=int(input("Enter the coloumn size"))
matrix=np.ndarray(shape=(r,c),dtype=int)
print("Enter the %d element in %dx%d matrix:" %(r*c,r,c))
for i in range(r):
    for j in range(c):
        matrix[i][j]=int(input())
print("matrix %dx%d is:" %(r,c))
print(matrix)
print("Sum of matrix is:",matrix.sum())


##5.Write a Python program to check whether a given number is a narcissistic number or not:

num=int(input("Enter the number"))
sum=0
temp=num
while num!=0:
    rem=num%10
    sum=sum+(rem*rem*rem)
    num=num//10

    print("Number is narcissistic")
else:
